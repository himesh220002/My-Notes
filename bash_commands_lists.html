1️⃣ MySQL Service Management

# Start MySQL service
sudo systemctl start mysql

# Stop MySQL service
sudo systemctl stop mysql

# Restart MySQL service
sudo systemctl restart mysql

# Check status
sudo systemctl status mysql

# View logs
sudo journalctl -xeu mysql.service --no-pager

Function: Manage MySQL server state and debug startup issues.

2️⃣ Checking Processes

# List all MySQL processes
ps aux | grep mysqld

# Filter specifically for MySQL daemon
ps -ef | grep mysqld

Function: Verify if MySQL server or safe mode is running.

3️⃣ MySQL Safe Mode (skip password)

# Start MySQL without password checking
sudo mysqld_safe --skip-grant-tables --skip-networking

Function: Bypass authentication to reset root password.

4️⃣ File & Directory Permissions

# Create MySQL runtime directory
sudo mkdir -p /var/run/mysqld

# Set ownership
sudo chown mysql:mysql /var/run/mysqld

# Check error log
sudo less /var/log/mysql/error.log

Function: Fix socket or log file issues that prevent MySQL from starting.

5️⃣ MySQL CLI Commands

-- Switch to mysql database
USE mysql;

-- Reset root password
UPDATE user
SET authentication_string = NULL,
    plugin = 'mysql_native_password'
WHERE User='root' AND Host='localhost';

-- Set new password
ALTER USER 'root'@'localhost' IDENTIFIED BY 'YourNewStrongPassword123!';

-- Apply changes
FLUSH PRIVILEGES;

-- Exit CLI
EXIT;

Function: Manage MySQL users and reset root password.

6️⃣ General Utilities

# Check running processes
ps aux

# Search for a process
ps aux | grep <process_name>

# View last lines of a log file
sudo tail -n 50 /var/log/mysql/error.log

Function: System process and log inspection.
